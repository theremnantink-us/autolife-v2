# Redesign (Phase 2) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Переписать фронтенд autolife-detail.ru на Astro + React islands с Spline-hero, гибридным routing (`/` + `/services/[slug]`), mobile-first подходом и mini-PHP endpoint для CSRF. PHP-бэкенд (api.php, busy_dates.php, health.php, lib/*) остаётся без изменений.

**Architecture:** Astro 4 как статический генератор в `site/`. React 18 как единственный island-framework. Интерактивные точки (Spline hero, форма записи, галерея, отзывы) — React-компоненты с селективной hydration (`client:visible`, `client:load`, `client:media`). Build → `site/dist/` rsync'ится в `public_html/dist/`. `.htaccess` rewrite направляет не-PHP трафик в `dist/`.

**Tech Stack:** Astro 4+, React 18, motion 12, @splinetool/react-spline, react-hook-form, zod, flatpickr, vitest (unit), playwright (e2e smoke). Styling: CSS Modules + brand tokens из Phase 1 в `:root`.

**Spec:** [docs/superpowers/specs/2026-04-24-redesign-design.md](../specs/2026-04-24-redesign-design.md)

```ts
// site/src/lib/csrf.ts
export async function fetchCsrfToken(): Promise<string> {
  const res = await fetch('/csrf.php');
  if (!res.ok) {
    throw new Error(`CSRF endpoint returned ${res.status}`);
  }
  const data = await res.json();
  if (typeof data?.csrf_token !== 'string') {
    throw new Error('CSRF response has no csrf_token field');
  }
  return data.csrf_token;
}
```

```ts
export interface BookingPayload {
  name: string;
  phone: string;
  carBrand: string;
  carModel: string;
  service: string;

  message?: string;
  appointment_id?: number;
}
```

```ts
// site/src/lib/api.ts
import type { BookingPayload, BookingResponse, BusyDatesResponse } from './types';

export async function submitBooking(payload: BookingPayload): Promise<BookingResponse> {
  try {
    const res = await fetch('/api.php', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(payload),
    });
    return await res.json();
  } catch {
    return { success: false, message: 'Сеть недоступна. Попробуйте ещё раз.' };
  }
}

export async function fetchBusyDates(): Promise<string[]> {
  try {
    const res = await fetch('/busy_dates.php');
    if (!res.ok) return [];
    const data: BusyDatesResponse = await res.json();
    return Array.isArray(data.busy_dates) ? data.busy_dates : [];
  } catch {
    return [];
  }
}

export async function checkHealth(): Promise<boolean> {
  try {
    const res = await fetch('/health.php');
    if (!res.ok) return false;
    const data = await res.json();
    return data?.ok === true;
  } catch {
    return false;
  }
}
```

```tsx
// site/src/components/islands/ServicesGrid.tsx
import { motion } from 'motion/react';
import { useEffect, useState } from 'react';

interface ServiceCardData {
  slug: string;
  name: string;
  category: string;
  price_from: number;
  hero_image: string;
  short_description: string;
}

interface Props {
  services: ServiceCardData[];
}

const categoryOrder = ['Детейлинг', 'Мойка', 'Шиномонтаж'];

export default function ServicesGrid({ services }: Props) {
  const [filter, setFilter] = useState<string>('Все');
  const [visible, setVisible] = useState<ServiceCardData[]>(services);

  useEffect(() => {
    setVisible(filter === 'Все' ? services : services.filter(s => s.category === filter));
  }, [filter, services]);

  return (
    <section id="services" className="services-grid">
      <div className="container">
        <header className="services-grid__header">
          <p className="section-label">Услуги</p>
          <h2>Направления</h2>
          <div className="services-grid__filters" role="tablist">
            <button
              role="tab"
              aria-selected={filter === 'Все'}
              onClick={() => setFilter('Все')}
              className={filter === 'Все' ? 'active' : ''}
            >Все</button>
            {categoryOrder.map(cat => (
              <button
                key={cat}
                role="tab"
                aria-selected={filter === cat}
                onClick={() => setFilter(cat)}
                className={filter === cat ? 'active' : ''}
              >{cat}</button>
            ))}
          </div>
        </header>

        <motion.ul
          className="services-grid__list"
          initial="hidden"
          whileInView="visible"
          viewport={{ once: true, margin: '-100px' }}
          variants={{
            hidden: {},
            visible: { transition: { staggerChildren: 0.06 } },
          }}
        >
          {visible.map(service => (
            <motion.li
              key={service.slug}
              variants={{
                hidden: { opacity: 0, y: 20 },
```

```tsx
// site/src/components/islands/BookingForm.tsx
import { useEffect, useRef, useState } from 'react';
import { useForm } from 'react-hook-form';
import { z } from 'zod';
import flatpickr from 'flatpickr';
import 'flatpickr/dist/flatpickr.min.css';
import { submitBooking, fetchBusyDates } from '../../lib/api';
import { fetchCsrfToken } from '../../lib/csrf';

const BookingSchema = z.object({
  name: z.string().min(2, 'Имя должно содержать минимум 2 символа'),
  phone: z.string().refine(
    v => v.replace(/[^0-9+]/g, '').length >= 10,
    { message: 'Некорректный номер телефона' }
  ),
  carBrand: z.string().min(1, 'Укажите марку'),
  carModel: z.string().min(1, 'Укажите модель'),
  service: z.string().min(1, 'Выберите услугу'),
  date: z.string().min(1, 'Выберите дату и время'),
  additionalInfo: z.string().optional(),
});

type BookingFormData = z.infer<typeof BookingSchema>;

const SERVICES = [
  'Детейлинг мойка',
  'Полировка кузова',
  'Керамическое покрытие',
  'Химчистка кожаных сидений',
  'Химчистка салона',
  'Полировка фар',
  'Восстановление хрома',
  'Антидождь на стёкла',
  'Шиномонтаж',
];

export default function BookingForm() {
  const dateInputRef = useRef<HTMLInputElement>(null);
  const [csrfToken, setCsrfToken] = useState<string | null>(null);
  const [submitting, setSubmitting] = useState(false);
  const [submitResult, setSubmitResult] = useState<{ ok: boolean; message: string } | null>(null);

  const { register, handleSubmit, formState: { errors } } = useForm<BookingFormData>();

  // Fetch CSRF on mount
  useEffect(() => {
    fetchCsrfToken()
      .then(setCsrfToken)
      .catch(err => {
        console.error('CSRF fetch failed:', err);
        setCsrfToken(''); // Unblock form but submission will 403
      });
  }, []);

  // Init flatpickr + busy dates
  useEffect(() => {
    if (!dateInputRef.current) return;
    let instance: ReturnType<typeof flatpickr> | null = null;

    fetchBusyDates().then(busyDates => {
      instance = flatpickr(dateInputRef.current!, {
        enableTime: true,
        dateFormat: 'Y-m-d H:i',
        minDate: 'today',
        time_24hr: true,
        locale: { firstDayOfWeek: 1 } as any,
        minTime: '09:00',
        maxTime: '21:00',
        disable: [
          (date: Date) => date.getDay() === 0,
          ...busyDates,
        ],
      });
    });

    return () => { instance && (instance as any).destroy(); };
  }, []);

  const onSubmit = async (data: BookingFormData) => {
    const parsed = BookingSchema.safeParse(data);
    if (!parsed.success) return;

    setSubmitting(true);
    setSubmitResult(null);

    const result = await submitBooking({
      ...parsed.data,
      additionalInfo: parsed.data.additionalInfo ?? '',
      csrf_token: csrfToken ?? '',
    });

    setSubmitting(false);
    setSubmitResult({
      ok: result.success,
      message: result.success
        ? 'Запись успешно создана. Мы свяжемся с вами для подтверждения.'
        : result.message ?? 'Не удалось создать запись. Попробуйте позже.',
    });
  };

  return (
    <section id="booking" className="booking-form">
      <div className="container">
        <header className="booking-form__header">
          <p className="section-label">Запись</p>
          <h2>Запишитесь на услугу</h2>
        </header>

        <form onSubmit={handleSubmit(onSubmit)} className="booking-form__form" noValidate>
          <div className="form-row">
            <div className="form-field">
              <label htmlFor="name">Имя *</label>
              <input id="name" {...register('name')} disabled={csrfToken === null} />
              {errors.name && <span className="form-error">{errors.name.message}</span>}
            </div>
            <div className="form-field">
              <label htmlFor="phone">Телефон *</label>
              <input id="phone" type="tel" {...register('phone')} disabled={csrfToken === null} />
              {errors.phone && <span className="form-error">{errors.phone.message}</span>}
            </div>
          </div>

          <div className="form-row">
            <div className="form-field">
              <label htmlFor="carBrand">Марка *</label>
              <input id="carBrand" {...register('carBrand')} disabled={csrfToken === null} />
              {errors.carBrand && <span className="form-error">{errors.carBrand.message}</span>}
            </div>
            <div className="form-field">
              <label htmlFor="carModel">Модель *</label>
              <input id="carModel" {...register('carModel')} disabled={csrfToken === null} />
              {errors.carModel && <span className="form-error">{errors.carModel.message}</span>}
            </div>
          </div>

          <div className="form-field">
            <label htmlFor="service">Услуга *</label>
            <select id="service" {...register('service')} disabled={csrfToken === null}>
              <option value="">— Выберите —</option>
              {SERVICES.map(s => <option key={s} value={s}>{s}</option>)}
            </select>
            {errors.service && <span className="form-error">{errors.service.message}</span>}
          </div>

          <div className="form-field">
            <label htmlFor="date">Дата и время *</label>
            <input
              id="date"
              ref={dateInputRef}
              {...register('date')}
              placeholder="Выберите дату"
              disabled={csrfToken === null}
            />
            {errors.date && <span className="form-error">{errors.date.message}</span>}
          </div>

          <div className="form-field">
            <label htmlFor="additionalInfo">Комментарий</label>
            <textarea id="additionalInfo" rows={3} {...register('additionalInfo')} disabled={csrfToken === null} />
          </div>

          <button type="submit" className="btn-chrome" disabled={submitting || csrfToken === null}>
            {submitting ? 'Отправка…' : 'Записаться'}
          </button>

          {submitResult && (
            <div className={`form-result form-result--${submitResult.ok ? 'ok' : 'err'}`} role="status">
              {submitResult.message}
            </div>
          )}
        </form>
      </div>
    </section>
  );
}
```

```tsx
// site/src/components/islands/YandexReviews.tsx
import { useEffect, useState } from 'react';
import { fallbackReviews, type Review } from '../../content/reviews';

interface Props {
  yandexApiKey?: string;        // read from import.meta.env in parent; optional
  yandexPlaceId?: string;       // place_id в Yandex.Бизнес
}

export default function YandexReviews({ yandexApiKey, yandexPlaceId }: Props) {
  const [reviews, setReviews] = useState<Review[]>(fallbackReviews);
  const [source, setSource] = useState<'yandex' | 'fallback'>('fallback');

  useEffect(() => {
    if (!yandexApiKey || !yandexPlaceId) return;

    // Fetch Yandex Places API reviews
    const url = `https://search-maps.yandex.ru/v1/?text=&type=biz&lang=ru_RU&apikey=${yandexApiKey}&id=${yandexPlaceId}`;
    fetch(url)
      .then(r => r.ok ? r.json() : Promise.reject())
      .then(data => {
        // Shape: features[0].properties.CompanyMetaData.Reviews...
        // (Real parsing depends on API response — validated in Phase 7)
        const items = data?.features?.[0]?.properties?.CompanyMetaData?.reviews ?? [];
        if (items.length) {
          setReviews(items.map((r: any) => ({
            author: r.author ?? 'Клиент',
            car: r.car ?? '',
            location: r.location ?? 'Москва',
            rating: r.rating ?? 5,
            text: r.text,
            date: r.date,
          })));
          setSource('yandex');
        }
      })
      .catch(() => { /* fallback stays */ });
  }, [yandexApiKey, yandexPlaceId]);

  return (
    <section id="reviews" className="reviews">
      <div className="container">
        <header className="reviews__header">
          <p className="section-label">Отзывы</p>
          <h2>Что говорят клиенты</h2>
          {source === 'yandex' && <span className="reviews__badge">Яндекс</span>}
        </header>
        <ul className="reviews__list">
          {reviews.map((r, i) => (
            <li key={i} className="reviews__card">
              <div className="reviews__stars" aria-label={`${r.rating} из 5`}>
                {'★'.repeat(r.rating)}{'☆'.repeat(5 - r.rating)}
              </div>
              <p className="reviews__text">{r.text}</p>
              <footer>
                <span className="reviews__author">{r.author}</span>
                {r.car && <span className="reviews__car">· {r.car}</span>}
                {r.location && <span className="reviews__location">· {r.location}</span>}
              </footer>
            </li>
          ))}
        </ul>
      </div>
    </section>
  );
}
```

```tsx
// site/src/components/islands/ServiceBookingCTA.tsx
import { useState, useEffect } from 'react';
import { useForm } from 'react-hook-form';
import { z } from 'zod';
import { submitBooking } from '../../lib/api';
import { fetchCsrfToken } from '../../lib/csrf';

const CompactBookingSchema = z.object({
  name: z.string().min(2, 'Имя должно содержать минимум 2 символа'),
  phone: z.string().refine(
    v => v.replace(/[^0-9+]/g, '').length >= 10,
    { message: 'Некорректный номер телефона' }
  ),
});

type CompactFormData = z.infer<typeof CompactBookingSchema>;

interface Props { serviceName: string; }

export default function ServiceBookingCTA({ serviceName }: Props) {
  const [csrfToken, setCsrfToken] = useState<string | null>(null);
  const [submitting, setSubmitting] = useState(false);
  const [submitResult, setSubmitResult] = useState<{ ok: boolean; message: string } | null>(null);
  const { register, handleSubmit, formState: { errors } } = useForm<CompactFormData>();

  useEffect(() => {
    fetchCsrfToken()
      .then(setCsrfToken)
      .catch(() => setCsrfToken(''));
  }, []);

  const onSubmit = async (data: CompactFormData) => {
    const parsed = CompactBookingSchema.safeParse(data);
    if (!parsed.success) return;

    setSubmitting(true);
    setSubmitResult(null);

    const result = await submitBooking({
      name: parsed.data.name,
      phone: parsed.data.phone,
      carBrand: '—',
      carModel: '—',
      service: serviceName,
      date: new Date(Date.now() + 24 * 60 * 60 * 1000).toISOString().slice(0, 10) + ' 10:00',
      additionalInfo: 'Записаться на консультацию (форма со страницы услуги)',
      csrf_token: csrfToken ?? '',
    });

    setSubmitting(false);
    setSubmitResult({
      ok: result.success,
      message: result.success
        ? 'Спасибо. Мы свяжемся с вами в течение часа.'
        : result.message ?? 'Не удалось отправить. Попробуйте форму на главной.',
    });
  };

  return (
    <section id="booking-cta" className="service-cta">
      <div className="container">
        <div className="service-cta__inner">
          <div className="service-cta__text">
            <h2>Записаться на {serviceName.toLowerCase()}</h2>
            <p>Оставьте имя и телефон. Мы перезвоним для уточнения даты и времени.</p>
          </div>
          <form onSubmit={handleSubmit(onSubmit)} className="service-cta__form" noValidate>
            <div className="form-field">
              <label htmlFor="cta-name">Имя</label>
              <input id="cta-name" {...register('name')} disabled={csrfToken === null || submitting} />
              {errors.name && <span className="form-error">{errors.name.message}</span>}
            </div>
            <div className="form-field">
              <label htmlFor="cta-phone">Телефон</label>
              <input id="cta-phone" type="tel" {...register('phone')} disabled={csrfToken === null || submitting} />
              {errors.phone && <span className="form-error">{errors.phone.message}</span>}
            </div>
            <button type="submit" className="btn-chrome" disabled={csrfToken === null || submitting}>
              {submitting ? 'Отправка…' : 'Записаться'}
            </button>
          </form>
          {submitResult && (
            <div className={`form-result form-result--${submitResult.ok ? 'ok' : 'err'}`} role="status">
              {submitResult.message}
            </div>
          )}
        </div>
      </div>
    </section>
  );
}
```

### Task 21: Deploy script (rsync to Timeweb)

**Files:**
- Create: `scripts/deploy.sh`

- [ ] **Step 1: Create deploy.sh**

```bash
#!/usr/bin/env bash
# scripts/deploy.sh — build Astro site and rsync to Timeweb public_html/
#
# Usage:
#   ./scripts/deploy.sh                    # uses env vars TIMEWEB_HOST / TIMEWEB_USER
#   TIMEWEB_HOST=... TIMEWEB_USER=... ./scripts/deploy.sh
#
# Prerequisites:
#   - SSH key configured for $TIMEWEB_USER@$TIMEWEB_HOST
#   - rsync installed locally
#   - public_html/.env exists on remote with real secrets

set -euo pipefail

TIMEWEB_HOST="${TIMEWEB_HOST:-autolife-detail.ru}"
TIMEWEB_USER="${TIMEWEB_USER:?missing TIMEWEB_USER env var}"
REMOTE_PATH="${REMOTE_PATH:-/home/$TIMEWEB_USER/public_html}"

ROOT="$(cd "$(dirname "$0")/.." && pwd)"

echo "==> Building Astro site"
cd "$ROOT/site"
npm run build

echo "==> rsync site/dist/ → $TIMEWEB_USER@$TIMEWEB_HOST:$REMOTE_PATH/dist/"
rsync -avz --delete \
  --exclude='.DS_Store' \
  "$ROOT/site/dist/" \
  "$TIMEWEB_USER@$TIMEWEB_HOST:$REMOTE_PATH/dist/"

echo "==> rsync PHP files (skipping .env)"
rsync -avz \
  --include='*.php' \
  --include='lib/***' \
  --include='IMG/***' \
  --include='icons/***' \
  --include='.htaccess' \
  --include='.env.example' \
  --exclude='.env' \
  --exclude='*' \
  "$ROOT/public_html/" \
  "$TIMEWEB_USER@$TIMEWEB_HOST:$REMOTE_PATH/"

echo "==> Deploy complete"
echo "Verify: https://$TIMEWEB_HOST/"
```
